    class Solution {
  public int singleNonDuplicate(int[] nums) {
      if (nums.length == 1) return nums[0];
      int pre = -1;
      for (int i = 0; i < nums.length; i++) {
          int curNum = nums[i];
          if (i == nums.length - 1) {
              if (pre != nums[i]) {
                  return curNum;
              }
          }
          if (pre != curNum && curNum != nums[i+1]) {
              return curNum;
          }
          pre = curNum;
      }
      return pre;
  }
}
