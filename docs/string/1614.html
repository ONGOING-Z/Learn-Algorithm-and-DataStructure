Problem Info

给定一字符串，算出字符串中最深的括号层数。

Answer - 1

class Solution {
    public int maxDepth(String s) {
        int res = 0;
        int max = 0;
        for (int i = 0; i < s.length(); i++) {
            if ('(' == s.charAt(i)) {
                res++;
                if (max < res) max = res;
            } else if (')' == s.charAt(i)) {
                res--;
            }
        }
        return max;
    }
}
