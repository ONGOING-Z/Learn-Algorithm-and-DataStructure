Problem Info

给定一字符串text，求字符串中字符可以组成字符串"ballon"的个数。

Answer - 1

class Solution {
        public int maxNumberOfBalloons(String text) {
        int res = 0;
        if (text.length() < 7) return 0;
        Map<Character, Integer> charMap = new HashMap<>();
        for (int i = 0; i < text.length(); i++) {
            char ch = text.charAt(i);
            if (charMap.containsKey(ch)) {
                charMap.put(ch, charMap.get(ch) + 1);
            } else {
                charMap.put(ch, 1);
            }
        }

        int max = res;
        List<Integer> numList = new ArrayList<>();
        String target = "balloon";
        for (int i = 0; i < target.length(); i++) {
            char ch = target.charAt(i);
            if (charMap.get(ch) != null) {
                if (ch == 'l' || ch == 'o') {
                    numList.add(charMap.get(ch) / 2);
                } else {
                    numList.add(charMap.get(ch));
                }
            } else {
                numList.add(0);
            }
        }
        max = Collections.min(numList);
        return max;
    }
}

class Solution {
        public int maxNumberOfBalloons(String text) {
        int res = 0;
        if (text.length() < 7) return 0;
        int[] resArray = new int[5];
        for (int i = 0; i < text.length(); i++) {
            char ch = text.charAt(i);
            if (ch == 'b') {
                resArray[0]++;
            } else if (ch == 'a') {
                resArray[1]++;
            } else if (ch == 'l') {
                resArray[2]++;
            } else if (ch == 'o') {
                resArray[3]++;
            } else if (ch == 'n') {
                resArray[4]++;
            }
        }
        resArray[2] /= 2;
        resArray[3] /= 2;

        return Arrays.stream(resArray).min().getAsInt();
    }
}

Attention

  • 字符 l 和字符 o 要用两次
