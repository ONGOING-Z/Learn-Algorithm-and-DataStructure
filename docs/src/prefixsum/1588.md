class Solution {
    public int sumOddLengthSubarrays(int[] arr) {
        int n = arr.length;
        int[] sum = new int[n+1];
        int res = 0;

        for (int i = 1; i <= n; i++) {
            sum[i] = sum[i-1] + arr[i-1];
        }

        for (int start = 0; start < n; start++) {
            for (int length = 1; start + length <= n; length+=2) {
                int end = start + length - 1;
                res += sum[end + 1] - sum[start];
            }
        }

        return res;

    }
}
