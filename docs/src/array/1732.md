```java
class Solution {
    public int largestAltitude(int[] gain) {
        int res = 0;
        int curHeight = 0;
        for (int i = 0; i < gain.length; i++) {
            curHeight += gain[i];
            res = Math.max(res, curHeight);
        }
        return res;
    }
}
```
