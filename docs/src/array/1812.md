class Solution {
    public boolean squareIsWhite(String coordinates) {
        char ch = coordinates.charAt(0);
        int resOne = ch - 'a' + 1;
        char num = coordinates.charAt(1);
        int resTwo = Integer.valueOf(num + "");

        // odd
        if (resOne % 2 == 1) {
            // black first
            if (resTwo % 2 == 1) {
                return false;
            }
            return true;
        }

        if (resTwo % 2 == 1) {
            return true;
        }
        return false;
    }

}
